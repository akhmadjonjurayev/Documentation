static AsymmetricCipherKeyPair GenerateRsaKeyPair()
{
    var keyGen = new RsaKeyPairGenerator();
    keyGen.Init(new KeyGenerationParameters(
        new SecureRandom(),
        2048)); // 2048 bit

    return keyGen.GenerateKeyPair();
}

static byte[] RsaEncrypt(byte[] data, AsymmetricKeyParameter publicKey)
{
    var engine = new Pkcs1Encoding(new RsaEngine());
    engine.Init(true, publicKey);

    return engine.ProcessBlock(data, 0, data.Length);
}

static byte[] RsaDecrypt(byte[] cipher, AsymmetricKeyParameter privateKey)
{
    var engine = new Pkcs1Encoding(new RsaEngine());
    engine.Init(false, privateKey);

    return engine.ProcessBlock(cipher, 0, cipher.Length);
}

static byte[] SignData(byte[] data, AsymmetricKeyParameter privateKey)
{
    var signer = new RsaDigestSigner(new Sha256Digest());
    signer.Init(true, privateKey);

    signer.BlockUpdate(data, 0, data.Length);
    return signer.GenerateSignature();
}

static bool VerifySignature(
    byte[] data,
    byte[] signature,
    AsymmetricKeyParameter publicKey)
{
    var signer = new RsaDigestSigner(new Sha256Digest());
    signer.Init(false, publicKey);

    signer.BlockUpdate(data, 0, data.Length);
    return signer.VerifySignature(signature);
}
